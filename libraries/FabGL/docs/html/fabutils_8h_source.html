103 template <typename T>
104 T moveItems(T dest, T src, size_t n)
105 {
106  T pd = dest;
107  T ps = src;
108  if (pd != ps) {
109  if (ps < pd)
110  for (pd += n, ps += n; n--;)
111  *--pd = *--ps;
112  else
113  while (n--)
114  *pd++ = *ps++;
115  }
116  return dest;
117 }

279 template <typename T>
280 class Stack {
281 public:
282  Stack() : m_items(nullptr) { }
283  bool isEmpty() { return m_items == nullptr; }
284  void push(T const & value) {
285  m_items = new StackItem<T>(m_items, value);
286  }
287  T pop() {
288  if (m_items) {
289  StackItem<T> * iptr = m_items;
290  m_items = iptr->next;
291  T r = iptr->item;
292  delete iptr;
293  return r;
294  } else
295  return T();
296  }
297  int count() {
298  int r = 0;
299  for (auto i = m_items; i; i = i->next)
300  ++r;
301  return r;
302  }
303 private:
304  StackItem<T> * m_items;
305 };

312 template <typename ...Params>
313 struct Delegate {
314 
315  template <typename Func>
316  void operator=(Func f) {
317  m_closure = [] (void * func, const Params & ...params) -> void { (*(Func *)func)(params...); };
318  m_func = heap_caps_malloc(sizeof(Func), MALLOC_CAP_32BIT);
319  moveItems<uint32_t*>((uint32_t*)m_func, (uint32_t*)&f, sizeof(Func) / sizeof(uint32_t));
320  }
321 
322  ~Delegate() {
323  heap_caps_free(m_func);
324  }
325 
326  void operator()(const Params & ...params) {
327  if (m_func)
328  m_closure(m_func, params...);
329  }
330 
331 private:
332  void (*m_closure)(void * func, const Params & ...params);
333  void * m_func = nullptr;
334 };
